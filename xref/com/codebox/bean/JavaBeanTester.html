15  package com.codebox.bean;
16  
17  import java.lang.reflect.Modifier;
18  
19  import net.bytebuddy.ByteBuddy;
20  import net.bytebuddy.NamingStrategy;
21  import net.bytebuddy.description.modifier.Visibility;
22  import net.bytebuddy.description.type.TypeDescription;
23  import net.bytebuddy.dynamic.loading.ClassLoadingStrategy;
24  import net.bytebuddy.implementation.EqualsMethod;
25  import net.bytebuddy.implementation.HashCodeMethod;
26  import net.bytebuddy.implementation.SuperMethodCall;
27  import net.bytebuddy.implementation.ToStringMethod;
28  import net.bytebuddy.matcher.ElementMatchers;
29  
30  /**
31   * This helper class can be used to unit test the get/set/equals/canEqual/toString/hashCode methods of JavaBean-style
32   * Value Objects.
33   */
34  public enum JavaBeanTester {
35  
36      // Private Usage
37      ;
38  
39      /**
40       * Configure JavaBeanTester using Fluent API.
41       *
42       * @param <T>
43       *            the generic type
44       * @param clazz
45       *            the clazz
46       *
47       * @return A builder implementing the fluent API to configure JavaBeanTester
48       */
49      public static <T> JavaBeanTesterBuilder<T, ?> builder(final Class<T> clazz) {
50          // If class is final, use Object.class for comparison needs
51          if (Modifier.isFinal(clazz.getModifiers())) {
52              return new JavaBeanTesterBuilder<>(clazz, Object.class);
53          }
54  
55          // Build extension from class using byte buddy
56          Class<? extends T> loaded = new ByteBuddy().with(new NamingStrategy.AbstractBase() {
57              @Override
58              protected String name(TypeDescription superClass) {
59                  return "com.codebox.bean.Extended" + superClass.getSimpleName();
60              }
61          }).subclass(clazz).method(ElementMatchers.any()).intercept(SuperMethodCall.INSTANCE)
62                  .method(ElementMatchers.isEquals()).intercept(EqualsMethod.requiringSuperClassEquality())
63                  .method(ElementMatchers.isHashCode()).intercept(HashCodeMethod.usingSuperClassOffset())
64                  .method(ElementMatchers.isToString()).intercept(ToStringMethod.prefixedBySimpleClassName())
65                  .defineField("javabeanExtension", String.class, Visibility.PACKAGE_PRIVATE).make()
66                  .load(clazz.getClassLoader(), ClassLoadingStrategy.Default.WRAPPER).getLoaded();
67  
68          // Builder with proper extension class
69          return builder(clazz, loaded);
70      }
71  
72      /**
73       * Configure JavaBeanTester using Fluent API.
74       *
75       * @param <T>
76       *            the generic type
77       * @param <E>
78       *            the element type
79       * @param clazz
80       *            the clazz
81       * @param extension
82       *            the extension
83       *
84       * @return A builder implementing the fluent API to configure JavaBeanTester
85       */
86      public static <T, E> JavaBeanTesterBuilder<T, E> builder(final Class<T> clazz, final Class<E> extension) {
87          return new JavaBeanTesterBuilder<>(clazz, extension);
88      }
89  }
